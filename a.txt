  "react-evacuation-bfs-": {
    "prefix": "react-evacuation-bfs-",
    "body": [
      "function App(){",
      "  const grid=[['S','.','.'],['#','#','.'],['.','.','E']];",
      "",
      "  const bfs=()=>{",
      "    const rows=grid.length, cols=grid[0].length;",
      "    const queue=[[0,0,0]];",
      "    const visited=new Set(['0-0']);",
      "    const dirs=[[1,0],[-1,0],[0,1],[0,-1]];",
      "",
      "    while(queue.length){",
      "      const [r,c,d]=queue.shift();",
      "      if(grid[r][c]==='E') return d;",
      "      for(const [dr,dc] of dirs){",
      "        const nr=r+dr, nc=c+dc;",
      "        if(nr>=0&&nc>=0&&nr<rows&&nc<cols&&grid[nr][nc]!=='#'&&!visited.has(`${nr}-${nc}`)){",
      "          visited.add(`${nr}-${nc}`);",
      "          queue.push([nr,nc,d+1]);",
      "        }",
      "      }",
      "    }",
      "    return 'No Path';",
      "  };",
      "",
      "  return(",
      "    <div>",
      "      <h1>Evacuation System</h1>",
      "      <p>Minimum Steps: {bfs()}</p>",
      "    </div>",
      "  );",
      "}",
      "",
      "export default App;"
