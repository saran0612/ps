  "Q15 Metro Train BFS": {
    "prefix": "metro-bfs",
    "description": "Shortest path with ticket validation",
    "body": [
      "function minSteps(grid) {",
      "  const rows = grid.length;",
      "  const cols = grid[0].length;",
      "  const queue = [];",
      "  const visited = new Set();",
      "",
      "  let start;",
      "  for (let r = 0; r < rows; r++) {",
      "    for (let c = 0; c < cols; c++) {",
      "      if (grid[r][c] === 'S') start = [r, c, false];",
      "    }",
      "  }",
      "",
      "  queue.push([...start, 0]);",
      "",
      "  while (queue.length) {",
      "    const [r, c, ticket, steps] = queue.shift();",
      "    if (grid[r][c] === 'E') return steps;",
      "",
      "    const key = `${r},${c},${ticket}`;",
      "    if (visited.has(key)) continue;",
      "    visited.add(key);",
      "",
      "    for (let [dr, dc] of [[1,0],[-1,0],[0,1],[0,-1]]) {",
      "      const nr = r + dr;",
      "      const nc = c + dc;",
      "      if (nr>=0 && nc>=0 && nr<rows && nc<cols) {",
      "        const cell = grid[nr][nc];",
      "        if (cell === '#') continue;",
      "        if (cell === 'G' && !ticket) continue;",
      "        queue.push([nr, nc, ticket || cell==='T', steps+1]);",
      "      }",
      "    }",
      "  }",
      "  return -1;",
      "}"
    ]
